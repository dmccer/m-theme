使用方法

第一步：将 m-apollo-theme-base 项目，添加为您的 submodule

通常如下：

git submodule add git@code.dianpingoa.com:f2e/m-apollo-theme-base.git ./less/common

第二步：在您的 page.less 中，import base/_include.less 和您用到的 component less

通常如下：

@import './../common/base/include.less';
@import './../common/component/default-list.less';

第三步：编写 gulp task，使 page.less 编译为 css 到指定目录（src/css）

通常如下：

var less = require('gulp-less');

gulp.task('less', function () {
  gulp.src('./less/page/**/*.less')
    .pipe(less({
      paths: [path.join(__dirname, 'less/common/base'), path.join(__dirname, 'less/common/component')]
    }))
    .pipe(gulp.dest('src/css/'));

  gulp.src('./less/common/img/**/*.*')
    .pipe(gulp.dest('src/img'));
});

Javascript

var Picture = require('efte-picture');
var $pictureContainer = $('.picture-container');

// 创建实例时，须传入图片预览的容器
// 也可传入初始展示的缩略图 (image || [image])

// 缩略图两个来源：
// 1. 服务端传过来
// 2. 手机中选择的图片或拍的图片

// image 格式(服务端传过来)
// {
//    name: '',
//    thumb: '' (base64)
// }

// 或(手机中选择的图片或拍的图片)
// {
//    url: '',
//    fullSizeUrl: '' 
// }

// Picture 有两种模式：1. 预览模式；2. 编辑模式
// 添加按钮的文案,即 title
var picture = new Picture($pictureContainer, {
  editable: true,
  title: '添加照片'
});
// var picture = new Picture($pictureContainer, image);
// var picture = new Picture($pictureContainer, [image]);

// 展示前可以先清空之前展示的缩略图
picture.empty();

// 展示缩略图
picture.show(image);
// picture.show([image]);

// 上传添加的图片
picture.upload({
  url: '/file',
  complete: function (oks, errs) {
    alert('complte upload: \n上传成功:' + oks.length + '\n上传失败: ' + errs.length);
  },
  timeoutCallback: function (aborts) {
    alert('上传超时: ' + aborts.length);
  },
  fail: function (code, err) {}
});

// 类方法可以上传指定的图片
Picture.upload({
  images: imageBase64 || [imageBase64],
  url: '/file',
  complete: function (oks, errs) {},
  timeoutCallback: function () {}
});
